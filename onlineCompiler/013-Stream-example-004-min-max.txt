import java.util.*;
import java.util.stream.*;
import java.util.function.*;

class Artist {
    private String from;
    private String name;
    private int rate;   
    
    Artist (String name, String from, int rate) {
        this.name=name;
        this.from= from;
        this.rate=rate;
    }
    
    public String toString() {
        return "Artist name="+name+", from="+from+", rate="+rate;
    }
    
    public boolean isFrom(String from) {
    	return this.from.equals(from);
    }

    public String getName() {
    	return name;
    }
    
    public int getRate() {
    	return rate;
    }
     
    public static Collection<Artist> getArtists() { 
    	var a1 = new Artist("Adele", "London",10);
        var a2 = new Artist("Zak Abel", "London",7);
        var a3 = new Artist("Adam Ant", "London",19);
        var a4 = new Artist("Helene Agofroy","Paris",3);
        var a5 = new Artist("Leila Alaoui","Paris",20);
        
        return Arrays.asList(a1,a2,a3,a4,a5);
    }
    
}

class Main {
    public static void main(String[] args) {

        
        Collection<Artist> artists = Artist.getArtists();
        

		//min rate    
        var min=artists.stream()
			.min(Comparator.comparing(artist -> artist.getRate()));
        var max=artists.stream()
			.max(Comparator.comparing(artist -> artist.getRate()));
       	//result are Optional
        System.out.println("min rated ="+min.orElse(null)); 
        System.out.println("max rated ="+max.orElse(null)); 
        
        //Empty list and steam
        artists = new ArrayList<>();
        min=artists.stream()
			.min(Comparator.comparing(artist -> artist.getRate()));
        max=artists.stream()
			.max(Comparator.comparing(artist -> artist.getRate()));
       	//result are Optional
        System.out.println("min rated ="+min.orElse(null)); 
        System.out.println("max rated ="+max.orElse(null)); 
    }
}
