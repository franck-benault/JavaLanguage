import java.util.*;

class Artist {
    private String from;
    private String name;
    
    Artist (String name, String from) {
        this.name=name;
        this.from= from;
    }
    
    public String toString() {
        return "Artist name="+name+", from="+from;
    }
}

class Main {
    public static void main(String[] args) {
	System.out.println("Java version :"+System.getProperty("java.version"));

        var a1 = new Artist("Adele", "London");
        var a2 = new Artist("Zak Abel", "London");
        var a3 = new Artist("Adam Ant", "London");
        var a4 = new Artist("Helene Agofroy","Paris");
        var a5 = new Artist("Leila Alaoui","Paris");
        
        //to allowed
        //var a6;
        //var a7 = null;

        var artists = Arrays.asList(a1,a2);
        //class java.util.Arrays$ArrayList bof
        System.out.println(artists.getClass());
    }
}
