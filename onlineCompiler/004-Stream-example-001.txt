import java.util.*;

class Artist {
    private String from;
    private String name;
    
    Artist (String name, String from) {
        this.name=name;
        this.from= from;
    }
    
    public String toString() {
        return "Artist name="+name+", from="+from;
    }
    
    public boolean isFrom(String from) {
    	return this.from.equals(from);
    }
    
}

class Main {
    public static void main(String[] args) {
        var a1 = new Artist("Adele", "London");
        var a2 = new Artist("Zak Abel", "London");
        var a3 = new Artist("Adam Ant", "London");
        var a4 = new Artist("Helene Agofroy","Paris");
        var a5 = new Artist("Leila Alaoui","Paris");
        
        Collection<Artist> artists = Arrays.asList(a1,a2,a3,a4,a5);
        
		//count artist from London
        var count=0L;
        for(Artist artist : artists) {
        	if(artist.isFrom("London")) count++;
        }
        System.out.println("Loop Count artist from London "+count); 
        
        count=artists.stream()
        	.filter(artist -> {
            	System.out.println(artist);
            	return artist.isFrom("London");
            })
            .count();
        System.out.println("Stream Count artist from London "+count); 
        
    }
}
